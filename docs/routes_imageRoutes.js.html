routes/imageRoutes.js

/**
 * @module routes/imageRoutes
 * @requires express
 * @requires controllers/ImageController
 * @requires middlewares/multer
 * @requires middlewares/auth
 */
import express from "express";
import { removeBgImage } from "../controllers/ImageController.js";
import upload from "../middlewares/multer.js";
import authUser from "../middlewares/auth.js";


/**
 * Express router instance for image-related API endpoints
 * @type {express.Router}
 */

const imageRouter = express.Router();


/**
 * @description Processes an uploaded image to remove its background using AI-powered ClipDrop API.
 * Requires user authentication and consumes one credit per operation.
 * 
 * @middleware upload.single("image") - Handles single file upload with field name "image"
 * @middleware authUser - Validates JWT token and extracts clerkId
 * @controller removeBgImage - Processes background removal request
 * 
 * @param {Object} request - Express request object
 * @param {Object} request.headers - Request headers
 * @param {string} request.headers.token - JWT authentication token
 * @param {FormData} request.body - Multipart form data
 * @param {File} request.body.image - Image file to process (jpg, png, etc.)
 * 
 * @returns {Object} JSON response with processing result
 * @returns {boolean} returns.success - Operation success status
 * @returns {string} returns.message - Response message
 * @returns {string} [returns.resultImage] - Base64 encoded processed image (on success)
 * @returns {number} [returns.creditBalance] - Updated user credit balance (on success)
 * @returns {boolean} [returns.timerActive] - Credit timer status (on success)
 * @returns {Date} [returns.nextCreditAt] - Next credit refresh time (on success)
 * 
 * @example
 * // Request:
 * // POST /api/image/remove-bg
 * // Headers: { "token": "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..." }
 * // Form Data: { "image": <file> }
 * 
 * // Success Response (200):
 * {
 *   "success": true,
 *   "resultImage": "data:image/png;base64,iVBORw0KGgoAAAANSUhEUgAA...",
 *   "creditBalance": 4,
 *   "timerActive": true,
 *   "nextCreditAt": "2024-01-15T10:32:00.000Z",
 *   "message": "Background Removed"
 * }
 * 
 * // Error Response - No Credits (200):
 * {
 *   "success": false,
 *   "message": "No credit balance",
 *   "creditBalance": 0
 * }
 * 
 * // Error Response - User Not Found (200):
 * {
 *   "success": false,
 *   "message": "User not found"
 * }
 * 
 * // Error Response - No Token (200):
 * {
 *   "success": false,
 *   "message": "Not authorized login again"
 * }
 * 
 */
imageRouter.post('/remove-bg', upload.single("image"), authUser, removeBgImage)

export default imageRouter
